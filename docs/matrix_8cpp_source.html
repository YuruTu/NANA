54void Matrix::create(int rows,int cols)
55{
56#ifdef NANA_DEBUG
57 std::cout << "内存申请\n";
58#endif
59 m_rows = rows;
60 m_cols = cols;
61 int temp = m_cols % 32;
62 if (0 == temp)
63 m_step = m_cols;
64 else
65 m_step = m_cols + 32 - temp;
66 size_t nSize =static_cast<size_t> (m_rows)* m_step;
67 m_data = new NAFLOAT[nSize];
68 m_val = new NAFLOAT * [m_rows];
69 m_val[0] = m_data;
70 for(int i=1;i<m_rows;i++)
71 m_val[i] = m_val[i - 1] + m_step;
72}
