103void Matrix::LU(const Matrix& A, Matrix& L, Matrix& U) {
104 int rows = A.rows();
105 int cols = A.cols();
106 NA_Assert(rows == cols);
107 Matrix Q = A;
108 L.release();
109 L.create(rows, cols);
110 U.release();
111 U.create(rows, cols);
112 NAFLOAT** q = Q.getValPtr();
113 NAFLOAT** l = L.getValPtr();
114 NAFLOAT** u = U.getValPtr();
115
116 int i, j, k;
117 for (k = 0; k < rows - 1; ++k) {
118 NA_Assert(std::fabs(q[k][k]) > NA_EPS);
119 for (i = k + 1; i < rows; ++i) {
120 q[i][k] /= q[k][k];
121 for (j = k + 1; j < cols; ++j) {
122 q[i][j] -= (q[i][k] * q[k][j]);
123 }
124
125 }
126 }
127
128 for (i = 0; i < rows; ++i) {
129 for (j = 0; j < i; ++j) {
130 l[i][j] = q[i][j];
131 u[i][j] = 0.0;
132 }
133 l[i][i] = 1.0;
134 u[i][i] = q[i][i];
135 for (j = i + 1; j < cols; ++j) {
136 l[i][j] = 0.0;
137 u[i][j] = q[i][j];
138 }

142void Matrix::QR(const Matrix& A, Matrix& Q, Matrix& R) {
143 int rows = A.rows();
144 int cols = A.cols();
145 R = A;
146 NA_Assert(rows >= cols);
147 setMatrixEye(Q, rows);
148 int ncols, i, j, k;
149 if (rows == cols)
150 ncols = rows - 1;
151 else
152 ncols = cols;
153 double u, alpha, w, t;
154 NAFLOAT** a = R.getValPtr();
155 NAFLOAT** q = Q.getValPtr();
156 for (k = 0; k < ncols; ++k) {
157 u = 0.0;
158 for (i = k; i < rows; ++i) {
159 w = std::fabs(a[i][k]);
160 if (w > u)
161 u = w;
162 }
163 NA_Assert(u >= NA_EPS);
164 alpha = 0.0;
165 for (i = k; i < rows; ++i) {
166 t = a[i][k] / u;
167 alpha += (t * t);
168 }
169 if (a[k][k] > 0.0)
170 u = -u;
171 alpha = u * std::sqrtf(alpha);
172 NA_Assert(std::fabs(alpha) >= NA_EPS);
173 u = std::sqrtf(2.0 * alpha * (alpha - a[k][k]));
174 NA_Assert(!isNan(u));
175 a[k][k] = (a[k][k] - alpha) / u;
176
177 for (i = k + 1; i < rows; ++i)
178 a[i][k] /= u;
179
180 for (j = 0; j < rows; ++j) {
181 t = 0.0;
182 for (i = k; i < rows; ++i)
183 t += a[i][k] * q[i][j];
184 for (i = k; i < rows; ++i)
185 q[i][j] -= 2.0 * t * a[i][k];
186 }
187 for (j = k + 1; j < cols; ++j) {
188 t = 0.0;
189 for (i = k; i < rows; ++i)
190 t += a[i][k] * a[i][j];
191 for (i = k; i < rows; ++i)
192 a[i][j] -= 2.0 * t * a[i][k];
193 }
194 a[k][k] = alpha;
195 for (i = k + 1; i < rows; ++i)
196 a[i][k] = 0.0;
197 }
198 for (i = 0; i < ncols; ++i) {
199 for (j = i + 1; j < rows; ++j) {
200 t = q[i][j];
201 q[i][j] = q[j][i];
202 q[j][i] = t;
203 }
204 }
205}

207#define SIGN(a,b) ((b) >= 0.0 ? std::fabs(a) : -std::fabs(a))
208static NAFLOAT sqrarg;
209#define SQR(a) ((sqrarg=(a)) == 0.0 ? 0.0 : sqrarg*sqrarg)
210NAFLOAT pythag(NAFLOAT a, NAFLOAT b) {
211 NAFLOAT absa, absb;
212 absa = fabs(a);
213 absb = fabs(b);
214 if (absa > absb)
215 return absa * std::sqrt(1.0 + SQR(absb / absa));
216 else
217 return (absb == 0.0 ? 0.0 : absb * sqrt(1.0 + SQR(absa / absb)));
218}

227void Matrix::SVD( const Matrix & A, Matrix& U2, Matrix& W, Matrix& V) {
228 Matrix U = A;
229 int m = A.rows();
230 int n = A.cols();
231 U2 = Matrix(m, m);
232 V = Matrix(n, n);
233
234 NAFLOAT* w = (NAFLOAT*)malloc(n * sizeof(NAFLOAT));
235 NAFLOAT* rv1 = (NAFLOAT*)malloc(n * sizeof(NAFLOAT));
236
237 int32_t flag, i, its, j, jj, k, l, nm;
238 NAFLOAT anorm, c, f, g, h, s, scale, x, y, z;
239
240 g = scale = anorm = 0.0;
241 for (i = 0; i < n; i++) {
242 l = i + 1;
243 rv1[i] = scale * g;
244 g = s = scale = 0.0;
245 if (i < m) {
246 for (k = i; k < m; k++) scale += fabs(U.m_val[k][i]);
247 if (scale) {
248 for (k = i; k < m; k++) {
249 U.m_val[k][i] /= scale;
250 s += U.m_val[k][i] * U.m_val[k][i];
251 }
252 f = U.m_val[i][i];
253 g = -SIGN(sqrt(s), f);
254 h = f * g - s;
255 U.m_val[i][i] = f - g;
256 for (j = l; j < n; j++) {
257 for (s = 0.0, k = i; k < m; k++) s += U.m_val[k][i] * U.m_val[k][j];
258 f = s / h;
259 for (k = i; k < m; k++) U.m_val[k][j] += f * U.m_val[k][i];
260 }
261 for (k = i; k < m; k++) U.m_val[k][i] *= scale;
262 }
263 }
264 w[i] = scale * g;
265 g = s = scale = 0.0;
266 if (i < m && i != n - 1) {
267 for (k = l; k < n; k++) scale += fabs(U.m_val[i][k]);
268 if (scale) {
269 for (k = l; k < n; k++) {
270 U.m_val[i][k] /= scale;
271 s += U.m_val[i][k] * U.m_val[i][k];
272 }
273 f = U.m_val[i][l];
274 g = -SIGN(sqrt(s), f);
275 h = f * g - s;
276 U.m_val[i][l] = f - g;
277 for (k = l; k < n; k++) rv1[k] = U.m_val[i][k] / h;
278 for (j = l; j < m; j++) {
279 for (s = 0.0, k = l; k < n; k++) s += U.m_val[j][k] * U.m_val[i][k];
280 for (k = l; k < n; k++) U.m_val[j][k] += s * rv1[k];
281 }
282 for (k = l; k < n; k++) U.m_val[i][k] *= scale;
283 }
284 }
285 anorm = std::max(anorm, (std::fabs(w[i]) + std::fabs(rv1[i])));
286 }
287 for (i = n - 1; i >= 0; i--) { // Accumulation of right-hand transformations.
288 if (i < n - 1) {
289 if (g) {
290 for (j = l; j < n; j++) // Double division to avoid possible underflow.
291 V.m_val[j][i] = (U.m_val[i][j] / U.m_val[i][l]) / g;
292 for (j = l; j < n; j++) {
293 for (s = 0.0, k = l; k < n; k++) s += U.m_val[i][k] * V.m_val[k][j];
294 for (k = l; k < n; k++) V.m_val[k][j] += s * V.m_val[k][i];
295 }
296 }
297 for (j = l; j < n; j++) V.m_val[i][j] = V.m_val[j][i] = 0.0;
298 }
299 V.m_val[i][i] = 1.0;
300 g = rv1[i];
301 l = i;
302 }
303 for (i = std::min(m, n) - 1; i >= 0; --i) { // Accumulation of left-hand transformations.
304 l = i + 1;
305 g = w[i];
306 for (j = l; j < n; j++) U.m_val[i][j] = 0.0;
307 if (g) {
308 g = 1.0 / g;
309 for (j = l; j < n; j++) {
310 for (s = 0.0, k = l; k < m; k++) s += U.m_val[k][i] * U.m_val[k][j];
311 f = (s / U.m_val[i][i]) * g;
312 for (k = i; k < m; k++) U.m_val[k][j] += f * U.m_val[k][i];
313 }
314 for (j = i; j < m; j++) U.m_val[j][i] *= g;
315 }
316 else for (j = i; j < m; j++) U.m_val[j][i] = 0.0;
317 ++U.m_val[i][i];
318 }
319 for (k = n - 1; k >= 0; k--) { // Diagonalization of the bidiagonal form: Loop over singular values,
320 for (its = 0; its < 30; its++) { // and over allowed iterations.
321 flag = 1;
322 for (l = k; l >= 0; l--) { // Test for splitting.
323 nm = l - 1;
324 if ((NAFLOAT)(fabs(rv1[l]) + anorm) == anorm) { flag = 0; break; }
325 if ((NAFLOAT)(fabs(w[nm]) + anorm) == anorm) { break; }
326 }
327 if (flag) {
328 c = 0.0; // Cancellation of rv1[l], if l > 1.
329 s = 1.0;
330 for (i = l; i <= k; i++) {
331 f = s * rv1[i];
332 rv1[i] = c * rv1[i];
333 if ((NAFLOAT)(fabs(f) + anorm) == anorm) break;
334 g = w[i];
335 h = pythag(f, g);
336 w[i] = h;
337 h = 1.0 / h;
338 c = g * h;
339 s = -f * h;
340 for (j = 0; j < m; j++) {
341 y = U.m_val[j][nm];
342 z = U.m_val[j][i];
343 U.m_val[j][nm] = y * c + z * s;
344 U.m_val[j][i] = z * c - y * s;
345 }
346 }
347 }
348 z = w[k];
349 if (l == k) { // Convergence.
350 if (z < 0.0) { // Singular value is made nonnegative.
351 w[k] = -z;
352 for (j = 0; j < n; j++) V.m_val[j][k] = -V.m_val[j][k];
353 }
354 break;
355 }
356
357 NA_Assert(its != 29,"ERROR in SVD: No convergence in 30 iterations");
358 x = w[l]; // Shift from bottom 2-by-2 minor.
359 nm = k - 1;
360 y = w[nm];
361 g = rv1[nm];
362 h = rv1[k];
363 f = ((y - z) * (y + z) + (g - h) * (g + h)) / (2.0 * h * y);
364 g = pythag(f, 1.0);
365 f = ((x - z) * (x + z) + h * ((y / (f + SIGN(g, f))) - h)) / x;
366 c = s = 1.0; // Next QR transformation:
367 for (j = l; j <= nm; j++) {
368 i = j + 1;
369 g = rv1[i];
370 y = w[i];
371 h = s * g;
372 g = c * g;
373 z = pythag(f, h);
374 rv1[j] = z;
375 c = f / z;
376 s = h / z;
377 f = x * c + g * s;
378 g = g * c - x * s;
379 h = y * s;
380 y *= c;
381 for (jj = 0; jj < n; jj++) {
382 x = V.m_val[jj][j];
383 z = V.m_val[jj][i];
384 V.m_val[jj][j] = x * c + z * s;
385 V.m_val[jj][i] = z * c - x * s;
386 }
387 z = pythag(f, h);
388 w[j] = z; // Rotation can be arbitrary if z = 0.
389 if (z) {
390 z = 1.0 / z;
391 c = f * z;
392 s = h * z;
393 }
394 f = c * g + s * y;
395 x = c * y - s * g;
396 for (jj = 0; jj < m; jj++) {
397 y = U.m_val[jj][j];
398 z = U.m_val[jj][i];
399 U.m_val[jj][j] = y * c + z * s;
400 U.m_val[jj][i] = z * c - y * s;
401 }
402 }
403 rv1[l] = 0.0;
404 rv1[k] = f;
405 w[k] = x;
406 }
407 }
408
409 // sort singular values and corresponding columns of u and v
410 // by decreasing magnitude. Also, signs of corresponding columns are
411 // flipped so as to maximize the number of positive elements.
412 int32_t s2, inc = 1;
413 NAFLOAT sw;
414 NAFLOAT* su = (NAFLOAT*)malloc(m * sizeof(NAFLOAT));
415 NAFLOAT* sv = (NAFLOAT*)malloc(n * sizeof(NAFLOAT));
416 do { inc *= 3; inc++; } while (inc <= n);
417 do {
418 inc /= 3;
419 for (i = inc; i < n; i++) {
420 sw = w[i];
421 for (k = 0; k < m; k++) su[k] = U.m_val[k][i];
422 for (k = 0; k < n; k++) sv[k] = V.m_val[k][i];
423 j = i;
424 while (w[j - inc] < sw) {
425 w[j] = w[j - inc];
426 for (k = 0; k < m; k++) U.m_val[k][j] = U.m_val[k][j - inc];
427 for (k = 0; k < n; k++) V.m_val[k][j] = V.m_val[k][j - inc];
428 j -= inc;
429 if (j < inc) break;
430 }
431 w[j] = sw;
432 for (k = 0; k < m; k++) U.m_val[k][j] = su[k];
433 for (k = 0; k < n; k++) V.m_val[k][j] = sv[k];
434 }
435 } while (inc > 1);
436 for (k = 0; k < n; k++) { // flip signs
437 s2 = 0;
438 for (i = 0; i < m; i++) if (U.m_val[i][k] < 0.0) s2++;
439 for (j = 0; j < n; j++) if (V.m_val[j][k] < 0.0) s2++;
440 if (s2 > (m + n) / 2) {
441 for (i = 0; i < m; i++) U.m_val[i][k] = -U.m_val[i][k];
442 for (j = 0; j < n; j++) V.m_val[j][k] = -V.m_val[j][k];
443 }
444 }
445
446 // create vector and copy singular values
447 W = Matrix(std::min(m, n), 1, w);
448
449 // extract mxm submatrix U
450 U2.setMat(U.getMat(0, 0, m - 1, std::min(m - 1, n - 1)), 0, 0);
451
452 // release temporary memory
453 free(w);
454 free(rv1);
455 free(su);
456 free(sv);
457}

615static int GaussiaJordanInv(const Matrix& src, Matrix& dst) {
616 int rows = src.rows();
617 int cols = src.cols();
618 NA_Assert(rows == cols);
620 dst = src;
621 int i, j, k, temp;
622 NAFLOAT** data = dst.getValPtr();
623
624 std::vector<int> is(rows);
625 std::vector<int> js(cols);
627 double d, p;
628 for (k = 0; k < rows; ++k) {
629 d = 0.0;
630 for (i = k; i < rows; ++i) {
631 for (j = k; j < cols; ++j) {
632 p = std::fabs(data[i][j]);
633 if (p > d) {
634 d = p;
635 is[k] = i;
636 js[k] = j;
637 }
638 }
639 }
640 NA_Assert(d >= NA_EPS);
641 temp = is[k];
642 if (temp != k) {
643 for (j = 0; j < cols; ++j) {
644 p = data[k][j];
645 data[k][j] = data[temp][j];
646 data[temp][j] = p;
647 }
648 }
649 temp = js[k];
650 if (temp != k) {
651 for (i = 0; i < rows; ++i) {
652 p = data[i][k];
653 data[i][k] = data[i][temp];
654 data[i][temp] = p;
655 }
656 }
658 data[k][k] = 1.0 / data[k][k];
659
660 for (j = 0; j < cols; ++j) {
661 if (k == j)
662 continue;
663 data[k][j] *= data[k][k];
664 }
666 for (i = 0; i < rows; ++i) {
667 if (i == k)
668 continue;
669 for (j = 0; j < cols; ++j) {
670 if (j == k)
671 continue;
672 data[i][j] -= data[i][k] * data[k][j];
673 }
674 }
675 for (i = 0; i < rows; ++i) {
676 if (i == k)
677 continue;
678 data[i][k] *= (-data[k][k]);
679 }
680 }
681
682
683 for (k = rows - 1; k >= 0; --k) {
684 temp = js[k];
685 if (k != temp) {
686 for (j = 0; j < cols; ++j) {
687 p = data[k][j];
688 data[k][j] = data[temp][j];
689 data[temp][j] = p;
690 }
691 }
692 temp = is[k];
693 if (k != temp) {
694 for (i = 0; i < rows; ++i) {
695 p = data[i][k];
696 data[i][k] = data[i][temp];
697 data[i][temp] = p;
698 }
699 }
700 }
701
702
703 return RET_OK;
704}

717NAFLOAT Matrix::det() const {
719 NA_Assert(m_rows == m_cols);
720 double f, detVal, q, d;
721 int i, j, k, is, js;
722 int n = m_rows;
723 f = 1.0;
724 detVal = 1.0;
725 Matrix A = *this;
726 double** a = A.getValPtr();
727 for (k = 0; k < n - 1; ++k) {
728 q = 0.0;
729 for (i = k; i < n; ++i) {
730 for (j = k; j < n; ++j) {
731 d = std::fabs(a[i][j]);
732 if (d > q) {
733 q = d;
734 is = i;
735 js = j;
736 }
737 }
738 }
739
740 if (q < NA_EPS) {
741 return 0.0;
742 }
743 if (is != k) {
744 f = -f;
745 for (j = k; j < n; ++j) {
746 d = a[k][j];
747 a[k][j] = a[is][j];
748 a[is][j] = d;
749 }
750 }
751
752 if (js != k) {
753 f = -f;
754 for (i = k; i < n; ++i) {
755 d = a[i][js];
756 a[i][js] = a[i][k];
757 a[i][k] = d;
758 }
759 }
760
761 detVal *= a[k][k];
762 for (i = k + 1; i < n; ++i) {
763 d = a[i][k] / a[k][k];
764 for (j = k + 1; j < n; ++j) {
765 a[i][j] -= d * a[k][j];
766 }
767 }
768 }
769 detVal *= f * a[n - 1][n - 1];
770 return detVal;
771}

776 double q, d;
778 Matrix A = *this;

780 nn = std::min(m_rows, m_cols);
781 k = 0;
782 for (cycle = 0; cycle < nn; ++cycle) {
783 q = 0.0;
784 for (i = 1; i < m_rows; ++i) {
785 for (j = 1; j < m_cols; ++j) {
786 d = std::fabs(a[i][j]);
787 if (d > q) {
788 q = d;
789 is = i;
790 js = j;
791 }
792 }
793 }
794
795 if (q < NA_EPS)
796 return k;
797
798 ++k;
799 if (is != cycle) {
800 for (j = 1; j < m_cols; ++j) {
801 d = a[cycle][j];
802 a[cycle][j] = a[is][j];

811 }

815 for (j = cycle + 1; j < m_cols; ++j) {
816 a[i][j] -= d * a[cycle][j];
817 }
818 }

821 }
